Angka Istimewa = 4(IV), 9(IX), 40(XL), 90(XC), 400(CD), 900(CM)

==== pseudocode ====

STORE arabicNum TO [1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000]
STORE romanNum TO [I, IV, V, IX, X, XL, L, XC, C, CD, D, CM, M]

STORE num TO any value
STORE result to empty string
STORE i TO arabicNum length minus 1

LOOP i TO 0

  WHILE num MORE THAN EQUAL arabicNum i
    CALCULATE num MINUS arabicNum i
    INSERT romanNum i TO result
  END WHILE

  i MINUS 1
END LOOP

return result
